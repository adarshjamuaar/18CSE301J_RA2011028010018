import pandas as pd
import matplotlib.pyplot as plt

# Load data from CSV file
data = pd.read_csv('gdp11-21.csv')

# Set the columns in the desired order
data = data[['Item', 'Year', 'Q1', 'Q2', 'Q3', 'Q4']]

# Create a pivot table to group the data by year and sector
pivot_data = pd.pivot_table(data, values=['Q1', 'Q2', 'Q3', 'Q4'], index='Year', columns='Item')

# Create a bar graph using the pivot table
pivot_data.plot(kind='bar', stacked=True, figsize=(65, 30))

# Set the title and axis labels
plt.title('GDP Contribution by Sector', fontsize=16)
plt.xlabel('Year', fontsize=9)
plt.ylabel('GDP (in billions of rupees)', fontsize=16)

# Show the plot
plt.show()
